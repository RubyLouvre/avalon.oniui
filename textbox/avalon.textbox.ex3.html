切换禁用组件（ms-disabled来控制）

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>切换禁用组件</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                <div class="demo-show" ms-controller="demo3">
                        <form>
                            <input ms-widget="textbox" ms-disabled="disable" ms-duplex="txt" type="textbox"/>
                            <button ms-click="changeDis" type="button">切换禁用</button>

                            <p>输入框的值是： {{txt}}</p>
                        </form>
                    </div>

                    <script>
                        require(["textbox/avalon.textbox"], function() {
                            avalon.define("demo3", function(vm) {
                                vm.txt = "点击 “切换禁用” 按钮，来切换禁用输入域";
                                vm.disable = true;
                                vm.changeDis = function() {
                                    vm.disable = !vm.disable;
                                }
                            })
                            avalon.scan();
                        })
                    </script>
                </body>
                </html>

输入框的值是： {{txt}}
