图片轮播组件

自定义不显示左右切换箭头和底部圆形选择部件
自定义鼠标经过继续播放 
<!DOCTYPE html>
<html>
    <head>
        <title>carousel组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        <script src="../highlight/shCore.js"></script>
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script>
            require(["carousel/avalon.carousel"], function() {
                avalon.define("demo1", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placekitten.com/g/520/300", 
                        "http://placekitten.com/g/600/400",
                        "http://placekitten.com/g/520/320"],
                        alwaysShowArrow: false, //@param alwaysShowArrow 显示左右切换箭头
                        alwaysShowSelection: false //@param alwaysShowSelection 显示底部圆形切换部件
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.define("demo2", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placekitten.com/g/520/300", 
                        "http://placekitten.com/g/600/400",
                        "http://placekitten.com/g/520/320"],
                        hoverStop: false //@param autoSlide 鼠标经过停止播放
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body>
        <div class="wrapper">
        <h1>图片轮播组件</h1>
        <fieldset ms-controller="demo1">
            <legend>自定义不显示左右切换箭头和底部圆形选择部件</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
        <fieldset ms-controller="demo2">
            <legend>自定义鼠标经过继续播放</legend>
            <div ms-widget="carousel"></div>
        </fieldset>

        </pre>
    </div>
    </body>
