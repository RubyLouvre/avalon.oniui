                var vm = avalon.define("test", function(vm) {

                        avalon.getScript("test.js", function() {
                            console.log("success")
                            vm.time = + new Date()
                            change()
                        })
                    }
                    vm.loadAJAX = function() {
                        avalon.ajax({

                            dataType: "script",
                            data: {
                                page: 1,
                                name: 1
                            },
                            success: function() {
                                console.log("success")
                                vm.time = + new Date()
                                change()
                            }
                        })
                    }
                    vm.jsonData = []
                    vm.time = ""
                })
                avalon.scan()
            })
