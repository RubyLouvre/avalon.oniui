图标式下拉开关（miniswitch）

用options配置组件并设置duplex

select：{{d}}

        <html>
        <head>
            <title></title>
            <meta http-equiv="content-type" content="text/html;charset=utf-8" />
            <meta http-equiv="x-ua-compatible" content="ie=edge" />
            <script src="../avalon.js"></script>
            <script type="text/javascript">
                require('miniswitch/avalon.miniswitch.js', function() {
                    avalon.ready(function() {
                        //multiple select vmodel
                        avalon.define('miniswitch', function(vm) {
                            vm.$opts = {
                                switchOpt: [{
                                    value: 1,
                                    label : ' 配置启用',
                                    cls: 'g-icon-start',
                                    title: '启用',
                                    font: '&#xf111;'
                                },
                                    {
                                        value: 2,
                                        label : ' 配置暂停',
                                        cls: 'g-icon-pause',
                                        title: '暂停',
                                        font: '&#xf04c;'
                                    }],
                                value: 2
                            };
                            vm.d = 1;
                        });
                        avalon.scan();
                    });
                });
            </script>
        </head>
        <body>
        <div class="wrapper">
            <div ms-controller="miniswitch">
                <h2>图标式下拉开关（miniswitch）</h2>
                <p>用options配置组件并设置duplex</p>
                <select ms-widget="miniswitch,$,$opts" ms-duplex="d"></select>
                <p>select：{{d}}</p>
            </div>
        </div>
        </body>
        </html>
